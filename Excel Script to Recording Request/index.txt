<script>
let workbookData = null;
let allSheets = [];

const fileInput = document.getElementById('fileInput');
const dropZone = document.getElementById('dropZone');

fileInput.addEventListener('change', e => {
  if (e.target.files[0]) loadFile(e.target.files[0]);
});

dropZone.addEventListener('dragover', e => { e.preventDefault(); dropZone.classList.add('dragover'); });
dropZone.addEventListener('dragleave', () => dropZone.classList.remove('dragover'));
dropZone.addEventListener('drop', e => {
  e.preventDefault();
  dropZone.classList.remove('dragover');
  if (e.dataTransfer.files[0]) loadFile(e.dataTransfer.files[0]);
});

document.getElementById('clientId').addEventListener('input', updatePreview);
document.getElementById('outreachNum').addEventListener('change', updatePreview);

function loadFile(file) {
  const reader = new FileReader();
  reader.onload = e => {
    const wb = XLSX.read(e.target.result, { type: 'array' });
    workbookData = wb;
    allSheets = wb.SheetNames;

    dropZone.classList.add('loaded');
    document.getElementById('dropIcon').textContent = '✅';
    document.getElementById('dropTitle').textContent = file.name;
    document.getElementById('dropSub').textContent = `${allSheets.length} sheet(s) found: ${allSheets.join(', ')}`;

    buildSheetGrid();
    updateOutreachOptions();
    document.getElementById('configCard').style.display = '';
    document.getElementById('actionCard').style.display = '';
    updatePreview();
  };
  reader.readAsArrayBuffer(file);
}

function buildSheetGrid() {
  const grid = document.getElementById('sheetGrid');
  grid.innerHTML = '';
  allSheets.forEach((name, i) => {
    const item = document.createElement('div');
    item.className = 'sheet-item active';
    item.dataset.sheet = name;
    item.innerHTML = `<input type="checkbox" checked id="cb_${i}"> <label for="cb_${i}" style="cursor:pointer">${name}</label>`;
    item.querySelector('input').addEventListener('change', () => {
      item.classList.toggle('active', item.querySelector('input').checked);
    });
    grid.appendChild(item);
  });
}

function updateOutreachOptions() {
  const sel = document.getElementById('outreachNum');
  sel.innerHTML = '';
  allSheets.forEach((name, i) => {
    const opt = document.createElement('option');
    opt.value = i + 1;
    opt.textContent = i + 1;
    sel.appendChild(opt);
  });
}

function getPrefix() {
  const cid = document.getElementById('clientId').value.trim().replace(/\s+/g, '_').toLowerCase() || 'client';
  return `e_${cid}`;
}

function cleanTitle(title) {
  if (!title) return '';
  return title
    .replace(/\n/g, ' ')
    .replace(/\s*\(.*?\)\s*/g, '')
    .trim()
    .replace(/[^a-zA-Z0-9\s]/g, '')
    .replace(/\s+/g, '')
    .toLowerCase();
}

function updatePreview() {
  const prefix = getPrefix();
  const example = `${prefix}_introduction`;
  document.getElementById('prefixPreview').innerHTML =
    `Preview: <strong>${example}_tt</strong> · <strong>${example}_qt</strong> · <strong>${example}_r1</strong>`;
}

function getSelectedSheets() {
  const items = document.querySelectorAll('.sheet-item');
  const selected = [];
  items.forEach(item => {
    if (item.querySelector('input').checked) selected.push(item.dataset.sheet);
  });
  return selected;
}

function parseSheet(sheetName) {
  const ws = workbookData.Sheets[sheetName];
  const raw = XLSX.utils.sheet_to_json(ws, { header: 1, defval: null });

  let headerRowIdx = -1;
  for (let i = 0; i < raw.length; i++) {
    if (raw[i] && raw[i][0] === 'Title') { headerRowIdx = i; break; }
  }
  if (headerRowIdx === -1) return [];

  const headers = raw[headerRowIdx];
  const colIdx = {
    title: headers.indexOf('Title'),
    transitionText: headers.findIndex(h => h && h.toLowerCase().includes('transition')),
    prompt: headers.findIndex(h => h && h.toLowerCase() === 'prompt'),
    smsQuestion: headers.findIndex(h => h && h.toLowerCase().includes('sms question')),
    responseTitle: headers.findIndex(h => h && h.toLowerCase().includes('response title')),
    voiceResponse: headers.findIndex(h => h && (h.toLowerCase().includes('voice response') || h.toLowerCase() === 'response 1')),
    smsResponse: headers.findIndex(h => h && h.toLowerCase().includes('sms response')),
  };

  // If SMS Question not found, try col 3 (index 3)
  if (colIdx.smsQuestion === -1) colIdx.smsQuestion = 3;

  const rows = raw.slice(headerRowIdx + 1);
  const prefix = getPrefix();
  const results = [];
  let currentTitle = null;
  let responseCounter = 0;

  rows.forEach(row => {
    if (!row || row.every(c => c === null || c === '')) return;

    const title = row[colIdx.title];
    const transitionText = row[colIdx.transitionText];
    const prompt = row[colIdx.prompt];
    const smsQuestion = colIdx.smsQuestion >= 0 ? row[colIdx.smsQuestion] : null;
    const responseTitle = colIdx.responseTitle >= 0 ? row[colIdx.responseTitle] : null;
    const voiceResponse = colIdx.voiceResponse >= 0 ? row[colIdx.voiceResponse] : null;
    const smsResponse = colIdx.smsResponse >= 0 ? row[colIdx.smsResponse] : null;

    if (title) {
      currentTitle = title;
      responseCounter = 0;

      // .tt row — Transition Text
      if (transitionText) {
        results.push({
          recordingName: `${prefix}_${cleanTitle(title)}_tt`,
          recordingDetails: transitionText,
          smsText: null,
          type: 'tt'
        });
      }

      // .qt row — Prompt
      if (prompt) {
        results.push({
          recordingName: `${prefix}_${cleanTitle(title)}_qt`,
          recordingDetails: prompt,
          smsText: smsQuestion || null,
          type: 'qt'
        });
      } else if (!transitionText && smsQuestion) {
        // No prompt but has SMS question (SMS-only question)
        results.push({
          recordingName: `${prefix}_${cleanTitle(title)}_qt`,
          recordingDetails: smsQuestion,
          smsText: smsQuestion,
          type: 'qt'
        });
      }
    }

    // .r rows — responses (can appear on title row or continuation rows)
    if (responseTitle && (voiceResponse || smsResponse)) {
      responseCounter++;
      const content = voiceResponse || smsResponse || '';
      results.push({
        recordingName: `${prefix}_${cleanTitle(currentTitle || responseTitle)}_r${responseCounter}`,
        recordingDetails: content,
        smsText: smsResponse || null,
        type: 'r'
      });
    }
  });

  return results;
}

function generateData(previewOnly) {
  const selected = getSelectedSheets();
  if (!selected.length) {
    alert('Please select at least one sheet.');
    return;
  }

  let allRows = [];
  selected.forEach(name => {
    const parsed = parseSheet(name);
    allRows = allRows.concat(parsed);
  });

  if (previewOnly) {
    renderPreview(allRows);
    document.getElementById('downloadBtn').disabled = false;
  } else {
    downloadExcel(allRows);
  }
}

function renderPreview(rows) {
  const tt = rows.filter(r => r.type === 'tt').length;
  const qt = rows.filter(r => r.type === 'qt').length;
  const r  = rows.filter(r => r.type === 'r').length;

  const section = document.getElementById('previewSection');
  section.innerHTML = `
    <div class="stats" style="margin-top:16px">
      <div class="stat-pill stat-total">📊 Total: ${rows.length}</div>
      <div class="stat-pill stat-tt">🔵 .tt (transition): ${tt}</div>
      <div class="stat-pill stat-qt">🟢 .qt (prompt): ${qt}</div>
      <div class="stat-pill stat-r">🟡 .r (response): ${r}</div>
    </div>
    <div class="preview-table-wrap">
      <table>
        <thead>
          <tr>
            <th style="width:220px">Recording Name</th>
            <th style="width:40px">Type</th>
            <th>Recording Details</th>
            <th style="width:220px">SMS Text</th>
          </tr>
        </thead>
        <tbody>
          ${rows.map(row => `
            <tr>
              <td class="code-cell">${esc(row.recordingName)}</td>
              <td><span class="tag tag-${row.type}">.${row.type}</span></td>
              <td class="sms-cell">${esc(row.recordingDetails || '')}</td>
              <td class="sms-cell">${row.smsText ? esc(row.smsText) : '<span style="color:#c5c8d6">—</span>'}</td>
            </tr>
          `).join('')}
        </tbody>
      </table>
    </div>
  `;
}

function esc(str) {
  return String(str || '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function downloadExcel(rows) {
  const wsData = [['Recording Name', 'Recording Details', 'Translations']];
  rows.forEach(row => {
    wsData.push([row.recordingName, row.recordingDetails || '', row.smsText || '']);
  });

  const wb = XLSX.utils.book_new();
  const ws = XLSX.utils.aoa_to_sheet(wsData);

  // Column widths
  ws['!cols'] = [{ wch: 45 }, { wch: 80 }, { wch: 80 }];

  // Style header row (basic)
  const headerStyle = { font: { bold: true } };
  ['A1','B1','C1'].forEach(ref => {
    if (!ws[ref].s) ws[ref].s = {};
    ws[ref].s = headerStyle;
  });

  XLSX.utils.book_append_sheet(wb, ws, 'Recording Requests');

  const cid = document.getElementById('clientId').value.trim().replace(/\s+/g,'_').toLowerCase() || 'client';
  XLSX.writeFile(wb, `recording_request_${cid}.xlsx`);
}

function resetAll() {
  workbookData = null;
  allSheets = [];
  document.getElementById('fileInput').value = '';
  document.getElementById('dropZone').classList.remove('loaded');
  document.getElementById('dropIcon').textContent = '📂';
  document.getElementById('dropTitle').textContent = 'Click or drag & drop your script .xlsx file';
  document.getElementById('dropSub').innerHTML = 'Supports <strong>CHF Outreach</strong> and similar CipherHealth script formats';
  document.getElementById('configCard').style.display = 'none';
  document.getElementById('actionCard').style.display = 'none';
  document.getElementById('previewSection').innerHTML = '';
}
